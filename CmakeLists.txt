cmake_minimum_required(VERSION 3.15)
project(MosaicEngine LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

if(MSVC)
    add_compile_options(/W4 /O2)
else()
    add_compile_options(-Wall -Wextra -O3 -march=native)
endif()

if(APPLE)
    list(APPEND CMAKE_PREFIX_PATH "/opt/homebrew")
    include_directories(/opt/homebrew/include)
    link_directories(/opt/homebrew/lib)
endif()

find_package(SDL2 REQUIRED)

if(NOT WIN32 AND NOT SDL2_FOUND)
    execute_process(COMMAND sdl2-config --cflags OUTPUT_VARIABLE SDL2_CFLAGS OUTPUT_STRIP_TRAILING_WHITESPACE)
    execute_process(COMMAND sdl2-config --libs OUTPUT_VARIABLE SDL2_LIBS OUTPUT_STRIP_TRAILING_WHITESPACE)
    separate_arguments(SDL2_CFLAGS_LIST UNIX_COMMAND "${SDL2_CFLAGS}")
endif()

set(SOURCES
    src/api/command_buffer.cpp
    src/core/device_executor.cpp
    src/main.cpp
)

add_executable(Mosaic ${SOURCES})

if(SDL2_FOUND)
    target_link_libraries(Mosaic PRIVATE SDL2::SDL2)
else()
    target_compile_options(Mosaic PRIVATE ${SDL2_CFLAGS_LIST})
    target_link_libraries(Mosaic PRIVATE ${SDL2_LIBS})
endif()
